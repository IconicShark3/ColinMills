def convertToBinary(n):
    thelist = []
    while n > 0:
        if n % 2 == 0:
            thelist.append(0)
            n = n / 2
        else:
            thelist.append(1)
            n = n / 2
            n = n - 0.5
    x = len(thelist) - 1
    final = str(thelist[x])
    x = x - 1
    while x >= 0:
        final = final + str(thelist[x])
        x = x - 1
    print(final)
